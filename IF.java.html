public class IF {
    public static void main(String[] args) {
        double height = 7.0;
        if (height >= 6.0) {
            System.out.println("Tall");
        }
        else if (height >= 5.7 && height < 6.0) {
            System.out.println("Average");
        }
        else {
            System.out.println("Short");
        }
    }
}
