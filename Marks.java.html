public class Marks {
    public static void main(String[] args) {

        int[][] numbers = {
                {11, 85, 90},
                {67, 72, 80},
                {88, 91, 79}
        };

        for (int i = 0; i < numbers.length; i++) {
            System.out.print("Student " + (i + 1) + ": ");
            for (int j = 0; j < numbers[i].length; j++) {
                System.out.print(numbers[i][j] + " ");
            }
            System.out.println();
        }
    }
}
